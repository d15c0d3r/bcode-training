class Queue{
    constructor(){
        this.q = []
    }
    enqueue(element){
        this.q.push(element)
    }
    dequeue(){
        this.q.shift()
    }
    front(){
        this.q[0]
    }
    isEmpty(){
        return !this.q.length > 0
    }
    displayQueue(){
        console.log(this.q)
    }
}
